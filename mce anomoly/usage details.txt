@app.route("/api/debug-hosts")
def debug_hosts():
    """List all available hosts and floors in the database"""
    try:
        query = """
SELECT DISTINCT host, floor_name
FROM "cpu"
WHERE customer_name='MCE'
        """
        result = client.query(query)
        hosts = []
        for (measurement, tags), points in result.items():
            for point in points:
                hosts.append({
                    "host": tags.get("host"),
                    "floor": tags.get("floor_name")
                })
        return jsonify({"hosts": hosts})
    except Exception as e:
        return jsonify({"error": str(e)}), 500

# API Endpoint for Graph Data with Predictions
# -------------------------------------------------
@app.route("/api/graph-data/<host>/<floor>")
def get_graph_data(host, floor):
    """Get live CPU usage data with anomaly predictions for a specific host/floor"""
    
    try:
        # Query live/recent CPU data from the last 10 minutes
        query = """
SELECT last("usage_idle") AS usage_idle
FROM "cpu"
WHERE customer_name='MCE' AND host='{}' AND floor_name='{}' 
        """.format(host, floor)
        
        print(f"Query: {query}")
        result = client.query(query)
        print(f"Query result: {result}")
        
        rows = []
        
        for (measurement, tags), points in result.items():
            for point in points:
                cpu_idle = point.get("usage_idle")
                if cpu_idle is None:
                    continue
                
                cpu_usage = 100 - cpu_idle
                timestamp = point.get("time")
                
                rows.append({
                    "timestamp": timestamp,
                    "cpu_usage": cpu_usage
                })
        
        print(f"Rows found: {len(rows)}")
        
        if not rows:
            return jsonify({"error": f"No CPU data found for host={host}, floor={floor}"}), 404
        
        # Sort by timestamp ascending
        df = pd.DataFrame(rows)
        df = df.sort_values('timestamp').reset_index(drop=True)
        
        # Prepare features for prediction
        df_feat = prepare_features(df)
        
        feature_cols = [
            "cpu_usage",
            "rolling_mean_5",
            "rolling_std_5",
            "rolling_max_5",
            "delta_usage",
            "z_score"
        ]
        
        # Fill NaN values with 0 for prediction
        df_feat[feature_cols] = df_feat[feature_cols].fillna(0)
        
        # Make predictions
        X = df_feat[feature_cols]
        df_feat["anomaly"] = model.predict(X)
        
        # Prepare response data
        graph_data = {
            "host": host,
            "floor": floor,
            "data": []
        }
        
        for _, row in df_feat.iterrows():
            graph_data["data"].append({
                "time": row["timestamp"],
                "cpu_usage": round(row["cpu_usage"], 2),
                "is_anomaly": int(row["anomaly"])
            })
        
        return jsonify(graph_data)
    
    except Exception as e:
        print(f"Error in get_graph_data: {str(e)}")
        import traceback
        traceback.print_exc()
        return jsonify({"error": str(e)}), 500
